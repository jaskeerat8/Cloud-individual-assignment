import random

p1, p2, p3 = 1/4, 1/2, 1/4
total, b1, b2, b3 = 10, 2, 5, 3
runs = 10000

success_count = 0
for i in range(runs):
    box_counts = [0, 0, 0]
    for j in range(total):
        rand = random.random()
        if(rand < p1):
            box_counts[0] = box_counts[0] + 1
        elif(rand < p1 + p2):
            box_counts[1] = box_counts[1] + 1
        else:
            box_counts[2] = box_counts[2] + 1

    if(box_counts == [b1, b2, b3]):
        success_count = success_count + 1

avg_prob = success_count / runs

print("Probability:", avg_prob)

Probability: 0.0762

from statistics import mean

total = 10
p2, p3 = 1/2, 1/4
iterations = 1000

list_prob = []
for i in range(iterations):
    probability = (p2 + p3) ** total
    list_prob.append(probability)

print("Probability that box 1 is empty:", mean(list_prob))

import numpy as np
import matplotlib.pyplot as plt

def pdf(x):
    return np.exp(-x) / (1 + np.exp(-x))**2

plt.plot(np.linspace(-10, 10, 10000), pdf(np.linspace(-10, 10, 10000)))
plt.title("PDF following Logistic Distribution")
plt.xlabel("x")
plt.ylabel("f(x)")
plt.show()
